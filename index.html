let versus var

var is quicksand

var cat = 'louis';
var i;

for (i = 1; (i <= 10); i++) {
  var cat = 'loopy cat #' + i;
  console.log(cat);
}
console.log(cat); // cat #10, not louis!

let: so scope-y fresh

let cat = 'louis';

for (let i = 1; (i <= 10); i++) {
  let cat = 'cat #' + i;
  console.log(cat);
}
console.log(cat); // louis the cat came back!

const makes you think

Sloppy sloppy slop

class Cat {
  constructor(name) {
    this.name = name;
  }
  pet() {
    console.log(this.name + ' purrs.');
  }
}
          
let cat;

cat = new Cat('Louis');
cat.pet();

cat = new Cat('SpySpy');
// I have to read the previous line
// to know which cat this refers to
cat.pet();

// More than one Louis? Surely you jest!
cat = new Cat('Louis');

class Cat {
  constructor(name) {
    this.name = name;
  }
  pet() {
    console.log(this.name + ' purrs.');
  }
}

const louis = new Cat('Louis');
louis.pet();

const spyspy = new Cat('SpySpy');
// I know which cat this is right away
spyspy.pet();

// This will fail
const louis = new Cat('Louis');

  const joined = messages.join(', ');

}
